pipeline PetalsWithLRPipeline;

// origem do dataset
dataset PetalsBase "Petals Dataset":
  from "path" with
    target as y = "targetColumnName"
    features as X = remaining
  preprocess all
    remove("column1Name")
    -> one_hot_encode("column2Name")
  visualize all {
    histogram,
    correlation
  }
  split all as h1
    train = 0.7,
    test = 0.2
  preprocess h1.train
    normalize("column2Name")
  preprocess h1.test
    normalize("column2Name")
